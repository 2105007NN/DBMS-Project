--TABLE FOR RECOMMANDATIONS ACCORDING TO SEARCHES
CREATE TABLE CUSTOMER_SEARCH(
	CUSTOMER_ID NUMBER,
	SEARCH_DATE TIMESTAMP,
	PRODUCT_ID NUMBER NOT NULL,
	CONSTRAINT SEARCH_PK PRIMARY KEY (CUSTOMER_ID, SEARCH_DATE),
	CONSTRAINT SEARCH_PRODUCT_FK FOREIGN KEY (PRODUCT_ID) REFERENCES PRODUCT(PRODUCT_ID)
	);
